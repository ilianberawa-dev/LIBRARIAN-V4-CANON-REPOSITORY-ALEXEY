--Cоздаем функцию для поиска по векторам документов: 

CREATE OR REPLACE FUNCTION public.match_documents(
    query_embedding vector, 
    match_count integer DEFAULT NULL::integer, 
    filter jsonb DEFAULT '{}'::jsonb
)
RETURNS TABLE(id bigint, content text, metadata jsonb, similarity double precision)
LANGUAGE plpgsql
AS $function$
BEGIN
    RETURN QUERY
    SELECT
        d.id,
        d.content,
        d.metadata,
        1 - (d.embedding <=> query_embedding) AS similarity
    FROM documents d
    WHERE TRUE
      -- Фильтрация по type, если он передан
      AND (
          filter ? 'type' IS FALSE
          OR d.metadata->>'type' = filter->>'type'
      )
      -- Фильтрация по file_id, если он передан
      AND (
          filter ? 'file_id' IS FALSE
          OR d.metadata->>'file_id' = filter->>'file_id'
      )
      
    ORDER BY d.embedding <=> query_embedding
    LIMIT COALESCE(match_count, 10);
END;
$function$;

-- Таблица с основной информацией о товарах
CREATE TABLE IF NOT EXISTS stocks_sup (
   id bigint PRIMARY KEY,
   name text,
   description text,
   category text,
   price numeric,
   in_stock text,
   rating numeric,
   review_count integer,
   sku text,
   manufacturer text,
   country text
);


-- Таблица с векторными представлениями товаров
CREATE TABLE IF NOT EXISTS stocks_sup_vector (
   id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
   content text,
   metadata jsonb,
   embedding vector
);

-- Добавляем колонку vector_id
ALTER TABLE stocks_sup
ADD COLUMN vector_id bigint;

-- Добавляем внешний ключ на vector
ALTER TABLE stocks_sup
ADD CONSTRAINT fk_vector_id
   FOREIGN KEY (vector_id)
   REFERENCES stocks_sup_vector(id)
   ON DELETE SET NULL;
 
 -- Удаляем запись из таблицы stocks_sup_vector, где id соответствует vector_id из таблицы stocks_sup
CREATE OR REPLACE FUNCTION delete_stocks_vector()
RETURNS trigger AS $$
BEGIN

 IF OLD.vector_id IS NOT NULL THEN
   DELETE FROM stocks_sup_vector WHERE id = OLD.vector_id;
 END IF;
 RETURN OLD;
END;
$$ LANGUAGE plpgsql;

-- Привязка триггера к таблице
create trigger trigger_delete_stocks_vector
after delete on stocks_sup
for each row
execute function delete_stocks_vector();
