--Cоздаем функцию для поиска по векторам документов: 

CREATE OR REPLACE FUNCTION public.match_documents(
    query_embedding vector, 
    match_count integer DEFAULT NULL::integer, 
    filter jsonb DEFAULT '{}'::jsonb
)
RETURNS TABLE(id bigint, content text, metadata jsonb, similarity double precision)
LANGUAGE plpgsql
AS $function$
BEGIN
    RETURN QUERY
    SELECT
        d.id,
        d.content,
        d.metadata,
        1 - (d.embedding <=> query_embedding) AS similarity
    FROM documents d
    WHERE TRUE
      -- Фильтрация по type, если он передан
      AND (
          filter ? 'type' IS FALSE
          OR d.metadata->>'type' = filter->>'type'
      )
      -- Фильтрация по file_id, если он передан
      AND (
          filter ? 'file_id' IS FALSE
          OR d.metadata->>'file_id' = filter->>'file_id'
      )
      
    ORDER BY d.embedding <=> query_embedding
    LIMIT COALESCE(match_count, 10);
END;
$function$;

docker run --rm -it -e TELEGRAM_API_ID=1231231123 -e TELEGRAM_API_HASH=2dkpokdkp884f29319cb736879ebb4 telegram-mcp  python utils/session_string_generator.py

cd leshiy

echo "th6btKVyeoTT" | docker login git.ak-devs.ru -u aa@bb.cc --password-stdin && chmod +x install.sh && ./install.sh

в echo пароль, например th6btKVyeoTT

Как это все работает на практике:

Вы даете задачу
                ✏️✏️✏️
Основной промпт анализирует эту задачу
                ✏️✏️✏️
Определяет нужного специалиста по use_case
               ✏️✏️✏️
Подгружает соответствующий промпт из базы
               ✏️✏️✏️
Специалист решает задачу своими инструментами

Почему такая структура работает:

✔️ Специализация: каждый агент решает узкий круг задач, но делает это на 100% (Насколько это возможно при текущем уровне развития нейросеток)
✔️ Чёткие use_case: система точно знает, какого эксперта вызывать для конкретной задачи
✔️ Правильные инструменты: каждый агент использует только те tools, которые ему действительно нужны

✔️ Подробные инструкции: внутри каждого промпта пошагово расписано ЧТО делать, КАК делать, и что делать НЕЛЬЗЯ

⚠️ Но есть один нюанс

Иногда ИИ, в независимости от модели( кто то больше, кто то меньше) всё равно забивает болт на эти инструкции и начинает импровизировать 🤦‍♂️

Наша задача по сути как оператора этой системы подтолкнуть его к правильному решению.

Неправильный подход:
"Исправь ошибку" → агент придумывает что-то от балды

Правильный подход:
"Дружище, ты делаешь не по инструкции. Во первых тебе нужно переключится в роль X. У тебя есть инструмент поиска по базе знаний, сходи туда, посмотри как правильно устроена работа с этой нодой/нодами, можешь помимо чтения документации изучить все примеры используемых нод из базы знаний, посмотреть готовые workflow c этой нодой, а так же дополнительно поищи инфу в инете и сделай как надо"

Практические фразы которые можно использовать:

🎯 "Стоп, ты отходишь от инструкций. Используй search_knowledge_base и найди актуальную информацию"
🎯 "Не выдумывай. У тебя есть get_workflow - сначала изучи структуру, потом предлагай решения"

🎯 "Ты игноришь инструменты. В инструкции чётко сказано использовать list_executions для анализа вот так и сделай"

💡 В чём фишка системы

Вместо одного "универсального" промпта на все случаи жизни, где он обязательно забьет хер на часть инструкций,  у нас команда узких "специалистов" под управлением главного координатора, над которым еще обязательно стоите ВЫ

2. Открой своего AI-агента (Claude Code / Cursor / Codex / Gemini / OpenClaw) в этой папке.

⚠️ Настоятельно рекомендую - не использовать основной аккаунт

Telegram может забанить аккаунт за API-активность (массовая отправка, парсинг, нетипичное поведение). Риск - потерять свой основной ТГ со всеми чатами и историей.

Что делать:
1. Зарегистрируй отдельный аккаунт под автоматизацию.
2. Номер - через сервисы аренды SMS: sms-reg.com и подобное. Выбирай страну, заказывай "Telegram", получай SMS, регистрируй новый аккаунт прямо в приложении ТГ на телефоне или в Telegram Desktop.
3. Этот аккаунт добавь в нужные чаты/группы руками - и только его используй со скиллом.
4. Даже если забанят - тебе не жалко, основной цел.

Авторизация (делается один раз):
Под уже созданным отдельным аккаунтом: зайди на https://my.telegram.org с этого номера → "API development tools" → создай приложение → получи api_id и api_hash. 

Запусти в терминале telegram auth, введи api_id, api_hash, номер, код из Telegram, 2FA если включил. 

Вообщем я позаботился чтобы агент вам сам все подсказал и правильно добавил 🙂

Ограничения: парсить, читать, писать может всё. Вступать в новые группы / подписываться на каналы из коробки не умеет (только то, где аккаунт уже состоит).
